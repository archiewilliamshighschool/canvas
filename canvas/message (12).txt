    <script>
        let balance = 100.00;
        let selectedNumbers = [];
        let drawnNumbers = [];
        let isPlaying = false;
        let currentPrize = 0;
        let currentDifficulty = 'easy';

        // Difficulty settings
        const difficulties = {
            easy: { ballsDrawn: 30, multiplier: 0.3 },
            medium: { ballsDrawn: 20, multiplier: 1.0 },
            hard: { ballsDrawn: 15, multiplier: 1.8 },
            impossible: { ballsDrawn: 'match', multiplier: 4.0 }
        };

        // Dynamic paytables based on numbers selected and difficulty
        const basePaytables = {
            1: { 1: 3.8 },
            2: { 1: 0.0, 2: 9.0 },
            3: { 1: 0.0, 2: 1.0, 3: 46.0 },
            4: { 1: 0.0, 2: 0.5, 3: 5.0, 4: 98.0 },
            5: { 2: 0.0, 3: 2.0, 4: 12.0, 5: 810.0 },
            6: { 3: 0.5, 4: 3.0, 5: 19.0, 6: 1600.0 },
            7: { 3: 0.5, 4: 1.0, 5: 7.0, 6: 400.0, 7: 7000.0 },
            8: { 4: 0.5, 5: 2.0, 6: 25.0, 7: 1550.0, 8: 25000.0 },
            9: { 4: 0.5, 5: 1.0, 6: 15.0, 7: 300.0, 8: 4700.0, 9: 40000.0 },
            10: { 5: 5.0, 6: 8.0, 7: 12.0, 8: 18.0, 9: 30.0, 10: 50.0 }
        };

        // Initialize the game
        function initGame() {
            createKenoBoard();
            updateBalance();
            updateSelectedCount();
            updatePaytable();
        }

        function createKenoBoard() {
            const board = document.getElementById('kenoBoard');
            board.innerHTML = '';
            
            for (let i = 1; i <= 80; i++) {
                const numberElement = document.createElement('div');
                numberElement.className = 'keno-number';
                numberElement.textContent = i;
                numberElement.onclick = () => toggleNumber(i);
                board.appendChild(numberElement);
            }
        }

        function toggleNumber(number) {
            if (isPlaying) return;
            
            const numberElement = document.querySelector(`.keno-number:nth-child(${number})`);
            
            if (selectedNumbers.includes(number)) {
                selectedNumbers = selectedNumbers.filter(n => n !== number);
                numberElement.classList.remove('selected');
            } else if (selectedNumbers.length < 10) {
                selectedNumbers.push(number);
                numberElement.classList.add('selected');
            }
            
            updateSelectedNumbers();
            updateSelectedCount();
            updateGameStatus();
        }

        function updateSelectedNumbers() {
            const selectedList = document.getElementById('selectedList');
            selectedList.innerHTML = '';
            
            selectedNumbers.sort((a, b) => a - b).forEach(number => {
                const chip = document.createElement('div');
                chip.className = 'selected-chip';
                chip.textContent = number;
                selectedList.appendChild(chip);
            });
        }

        function updateSelectedCount() {
            document.getElementById('selectedCount').textContent = selectedNumbers.length;
            updatePaytable();
        }

        function updateGameStatus() {
            const status = document.getElementById('gameStatus');
            if (selectedNumbers.length === 0) {
                status.textContent = 'Select 1-10 numbers and choose difficulty';
            } else if (selectedNumbers.length === 1) {
                status.textContent = `1 number selected (${currentDifficulty.toUpperCase()}) - Ready to play!`;
            } else {
                status.textContent = `${selectedNumbers.length} numbers selected (${currentDifficulty.toUpperCase()}) - Ready to play!`;
            }
            status.className = '';
        }

        function setDifficulty(difficulty) {
            if (isPlaying) return;
            
            currentDifficulty = difficulty;
            
            // Update button states
            document.querySelectorAll('.difficulty-btn').forEach(btn => {
                btn.classList.remove('active');
            });
            document.getElementById(difficulty + '-btn').classList.add('active');
            
            updateGameStatus();
            updatePaytable();
        }

        function updatePaytable() {
            const paytableContent = document.getElementById('paytableContent');
            
            if (selectedNumbers.length === 0) {
                paytableContent.innerHTML = '<div class="paytable-row"><span>Select numbers to see payouts</span><span>-</span></div>';
                return;
            }
            
            const numSelected = selectedNumbers.length;
            const basePaytable = basePaytables[numSelected];
            const difficultyMultiplier = difficulties[currentDifficulty].multiplier;
            
            let html = '';
            for (const [hits, multiplier] of Object.entries(basePaytable)) {
                let adjustedMultiplier = multiplier * difficultyMultiplier;
                
                // Format multiplier display
                let displayMultiplier;
                if (adjustedMultiplier === 0) {
                    displayMultiplier = "LOSS";
                } else if (adjustedMultiplier < 1) {
                    displayMultiplier = adjustedMultiplier.toFixed(2) + "x";
                } else if (adjustedMultiplier < 10) {
                    displayMultiplier = adjustedMultiplier.toFixed(1) + "x";
                } else {
                    displayMultiplier = Math.round(adjustedMultiplier) + "x";
                }
                
                html += `<div class="paytable-row">
                    <span>${hits} Hit${hits > 1 ? 's' : ''}:</span>
                    <span>${displayMultiplier}</span>
                </div>`;
            }
            
            paytableContent.innerHTML = html;
        }

        function setBet(amount) {
            document.getElementById('betAmount').value = amount;
        }

        function clearSelection() {
            if (isPlaying) return;
            
            selectedNumbers = [];
            drawnNumbers = [];
            currentPrize = 0;
            
            document.querySelectorAll('.keno-number').forEach(el => {
                el.classList.remove('selected', 'drawn', 'hit');
            });
            
            updateSelectedNumbers();
            updateSelectedCount();
            updateGameStatus();
            updatePrizeMeter(0);
            updatePaytable();
            
            document.getElementById('drawnList').innerHTML = '';
        }

        function playKeno() {
            if (isPlaying || selectedNumbers.length === 0) return;
            
            const betAmount = parseInt(document.getElementById('betAmount').value);
            if (betAmount > balance) {
                alert('Insufficient balance!');
                return;
            }
            
            // Deduct bet from balance
            balance -= betAmount;
            updateBalance();
            
            isPlaying = true;
            document.getElementById('playBtn').disabled = true;
            document.getElementById('playBtn').textContent = 'DRAWING...';
            
            const status = document.getElementById('gameStatus');
            status.textContent = 'Drawing numbers...';
            status.className = 'drawing-animation';
            
            // Clear previous drawn numbers
            drawnNumbers = [];
            document.getElementById('drawnList').innerHTML = '';
            document.querySelectorAll('.keno-number').forEach(el => {
                el.classList.remove('drawn', 'hit');
            });
            
            // Generate 20 random numbers
            const allNumbers = Array.from({length: 80}, (_, i) => i + 1);
            drawnNumbers = [];
            
            // Draw numbers with animation
            drawNumbersWithAnimation(allNumbers, betAmount);
        }

        function drawNumbersWithAnimation(allNumbers, betAmount) {
            const difficulty = difficulties[currentDifficulty];
            let totalBallsToDraw;
            
            if (difficulty.ballsDrawn === 'match') {
                totalBallsToDraw = selectedNumbers.length;
            } else {
                totalBallsToDraw = difficulty.ballsDrawn;
            }
            
            const drawInterval = setInterval(() => {
                if (drawnNumbers.length >= totalBallsToDraw) {
                    clearInterval(drawInterval);
                    finishGame(betAmount);
                    return;
                }
                
                // Pick a random number that hasn't been drawn
                const availableNumbers = allNumbers.filter(n => !drawnNumbers.includes(n));
                const randomIndex = Math.floor(Math.random() * availableNumbers.length);
                const drawnNumber = availableNumbers[randomIndex];
                
                drawnNumbers.push(drawnNumber);
                
                // Animate the number on the board
                const numberElement = document.querySelector(`.keno-number:nth-child(${drawnNumber})`);
                numberElement.classList.add('drawn');
                
                // Check if it's a hit
                const isHit = selectedNumbers.includes(drawnNumber);
                if (isHit) {
                    setTimeout(() => {
                        numberElement.classList.add('hit');
                        updatePrizeMeter(calculateCurrentPrize(betAmount));
                        addSparkles();
                    }, 200);
                }
                
                // Add to drawn list
                addToDrawnList(drawnNumber, isHit);
                
            }, 150);
        }

        function addToDrawnList(number, isHit) {
            const drawnList = document.getElementById('drawnList');
            const chip = document.createElement('div');
            chip.className = isHit ? 'drawn-chip hit' : 'drawn-chip';
            chip.textContent = number;
            drawnList.appendChild(chip);
        }

        function calculateCurrentPrize(betAmount) {
            const hits = drawnNumbers.filter(n => selectedNumbers.includes(n)).length;
            const numSelected = selectedNumbers.length;
            
            if (!basePaytables[numSelected] || !basePaytables[numSelected][hits]) {
                return 0;
            }
            
            const baseMultiplier = basePaytables[numSelected][hits];
            const difficultyMultiplier = difficulties[currentDifficulty].multiplier;
            const finalMultiplier = baseMultiplier * difficultyMultiplier;
            
            return betAmount * finalMultiplier;
        }

        function updatePrizeMeter(prize) {
            currentPrize = prize;
            const meterFill = document.getElementById('meterFill');
            const meterValue = document.getElementById('meterValue');
            
            const betAmount = parseInt(document.getElementById('betAmount').value);
            const numSelected = selectedNumbers.length;
            
            let maxPossiblePrize = 0;
            if (numSelected > 0 && basePaytables[numSelected]) {
                const maxBaseMultiplier = Math.max(...Object.values(basePaytables[numSelected]));
                const difficultyMultiplier = difficulties[currentDifficulty].multiplier;
                maxPossiblePrize = betAmount * maxBaseMultiplier * difficultyMultiplier;
            }
            
            const percentage = maxPossiblePrize > 0 ? Math.min((prize / maxPossiblePrize) * 100, 100) : 0;
            
            meterFill.style.height = percentage + '%';
            meterValue.textContent = `${prize.toFixed(2)}`;
            
            if (prize > 0) {
                meterValue.style.color = '#ffd700';
                meterValue.style.textShadow = '0 0 10px rgba(255, 215, 0, 0.8)';
            } else {
                meterValue.style.color = '#fff';
                meterValue.style.textShadow = '2px 2px 4px rgba(0, 0, 0, 0.8)';
            }
        }

        function addSparkles() {
            const meterContainer = document.querySelector('.meter-container');
            
            for (let i = 0; i < 5; i++) {
                setTimeout(() => {
                    const sparkle = document.createElement('div');
                    sparkle.className = 'meter-sparkle';
                    sparkle.style.left = Math.random() * 100 + '%';
                    sparkle.style.top = Math.random() * 100 + '%';
                    sparkle.style.animationDelay = Math.random() * 0.5 + 's';
                    meterContainer.appendChild(sparkle);
                    
                    setTimeout(() => sparkle.remove(), 1000);
                }, i * 100);
            }
        }

        function finishGame(betAmount) {
            const hits = drawnNumbers.filter(n => selectedNumbers.includes(n)).length;
            const finalPrize = calculateCurrentPrize(betAmount);
            
            // Add winnings to balance
            balance += finalPrize;
            updateBalance();
            
            // Update status
            const status = document.getElementById('gameStatus');
            if (hits === 0) {
                status.textContent = 'No matches - Better luck next time!';
                status.className = '';
            } else {
                status.textContent = `${hits} matches! You won $${finalPrize.toFixed(2)}!`;
                status.className = 'win-animation';
            }
            
            // Reset play button
            document.getElementById('playBtn').disabled = false;
            document.getElementById('playBtn').textContent = 'PLAY KENO';
            isPlaying = false;
        }

        function updateBalance() {
            document.getElementById('balance').textContent = balance.toFixed(2);
        }

        function goBack() {
            window.history.back();
        }

        // Initialize the game when page loads
        document.addEventListener('DOMContentLoaded', initGame);
